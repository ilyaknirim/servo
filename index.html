🔄 СЕРВОПРИВОД · 7 ЭКСПЕРИМЕНТОВ С ДВИЖЕНИЕМ

Сервопривод — это мотор, который умеет поворачиваться на заданный угол (обычно 0–180°) и точно его держать. Внутри: электромотор, редуктор и плата управления с обратной связью .

🔧 КАК РАБОТАЕТ СЕРВОПРИВОД

📐 Угол поворота

Стандартный сервопривод поворачивается на 180°. Управляется длительностью импульса: 0° = 0.5 мс, 90° = 1.5 мс, 180° = 2.5 мс .

⚡ Питание

Маленькие сервы (SG90, MG90S) работают от 5V, но при движении потребляют до 500 мА. Для одного серво — норм, для нескольких — нужно внешнее питание .

// ЭКСПЕРИМЕНТ 1: Базовое движение сервопривода // Плавно поворачивается от 0 до 180° и обратно #include <Servo.h> Servo myServo; // Создаём объект для управления сервоприводом int pos = 0; // Переменная для хранения текущего угла void setup() { myServo.attach(9); // Подключаем серво к 9 пину } void loop() { // Поворачиваемся от 0° до 180° for (pos = 0; pos <= 180; pos++) { myServo.write(pos); // Устанавливаем угол delay(15); // Ждём, чтобы серво успело повернуться } // Поворачиваемся обратно от 180° до 0° for (pos = 180; pos >= 0; pos--) { myServo.write(pos); delay(15); } }
🔬 ФАКТ О БИБЛИОТЕКЕ: Библиотека Servo автоматически генерирует импульсы нужной длительности. Для угла 0° — импульс 0.5 мс, для 180° — 2.5 мс . На пинах 9 и 10 при использовании серво нельзя использовать analogWrite() — они используют один таймер.
🕒 ПОЧЕМУ DELAY 15 МС? Сервопривод не мгновенно поворачивается — ему нужно время (~0.1–0.2 с на 60°). delay(15) на каждый градус даёт ~2.7 секунды на полный цикл. Можно увеличить скорость, уменьшив задержку.

// ЭКСПЕРИМЕНТ 2: Управление сервоприводом потенциометром // Крутим ручку — серво поворачивается #include <Servo.h> Servo myServo; #define POT_PIN A0 void setup() { myServo.attach(9); Serial.begin(9600); } void loop() { int potValue = analogRead(POT_PIN); // 0..1023 // Преобразуем 0..1023 в 0..180 int angle = map(potValue, 0, 1023, 0, 180); myServo.write(angle); Serial.print("Потенциометр: "); Serial.print(potValue); Serial.print(" → Угол: "); Serial.println(angle); delay(50); // Небольшая пауза }
🎛️ ТОЧНОЕ ПОЗИЦИОНИРОВАНИЕ: Так управляются промышленные манипуляторы — оператор крутит ручки, робот повторяет движения. В реальных системах используют не потенциометры, а энкодеры (цифровые датчики угла).

// ЭКСПЕРИМЕНТ 3: Два сервопривода // Один поворачивается влево-вправо (основание) // Второй — вверх-вниз (плечо) #include <Servo.h> Servo baseServo; // Серво основания Servo armServo; // Серво плеча void setup() { baseServo.attach(9); armServo.attach(10); } void loop() { // Основание поворачивается for (int angle = 0; angle <= 180; angle++) { baseServo.write(angle); delay(10); } // Плечо поднимается for (int angle = 0; angle <= 180; angle++) { armServo.write(angle); delay(10); } // Основание возвращается for (int angle = 180; angle >= 0; angle--) { baseServo.write(angle); delay(10); } // Плечо опускается for (int angle = 180; angle >= 0; angle--) { armServo.write(angle); delay(10); } }
🤖 РОБОТЕХНИКА: Два сервопривода — это уже полноценная SCARA-рука (Selective Compliance Assembly Robot Arm). Один отвечает за поворот (основание), второй — за подъём (плечо). Третий серво — предплечье, четвёртый — запястье.

// ЭКСПЕРИМЕНТ 4: Три сервопривода (манипулятор) // Движения по заданной траектории #include <Servo.h> Servo baseServo; // Основание (поворот) Servo shoulderServo; // Плечо (вверх-вниз) Servo elbowServo; // Предплечье int basePos = 90; int shoulderPos = 90; int elbowPos = 90; void setup() { baseServo.attach(9); shoulderServo.attach(10); elbowServo.attach(11); // Начальное положение baseServo.write(basePos); shoulderServo.write(shoulderPos); elbowServo.write(elbowPos); delay(1000); } void loop() { // Движение 1: поворот основания for (basePos = 90; basePos <= 180; basePos++) { baseServo.write(basePos); delay(10); } // Движение 2: подъём плеча for (shoulderPos = 90; shoulderPos <= 150; shoulderPos++) { shoulderServo.write(shoulderPos); delay(10); } // Движение 3: сгибание предплечья for (elbowPos = 90; elbowPos <= 120; elbowPos++) { elbowServo.write(elbowPos); delay(10); } delay(1000); // Возврат в исходное for (elbowPos = 120; elbowPos >= 90; elbowPos--) { elbowServo.write(elbowPos); delay(10); } for (shoulderPos = 150; shoulderPos >= 90; shoulderPos--) { shoulderServo.write(shoulderPos); delay(10); } for (basePos = 180; basePos >= 90; basePos--) { baseServo.write(basePos); delay(10); } delay(1000); }
🦾 КИНЕМАТИКА: В реальных роботах положение схвата вычисляется с помощью прямой и обратной кинематики. Для трёх звеньев это уже довольно сложная математика (матрицы поворота, углы Эйлера). Но принцип тот же — координация нескольких моторов.

// ЭКСПЕРИМЕНТ 5: Полноценная рука с запястьем // Кнопка записывает последовательность движений #include <Servo.h> Servo servos[4]; // Массив сервоприводов int pins[4] = {9, 10, 11, 6}; #define BUTTON 2 // Массив для записи до 100 позиций int positions[100][4]; int stepCount = 0; bool recording = false; void setup() { for (int i = 0; i < 4; i++) { servos[i].attach(pins[i]); servos[i].write(90); } pinMode(BUTTON, INPUT); Serial.begin(9600); Serial.println("Нажмите кнопку для записи"); } void loop() { // Проверяем кнопку if (digitalRead(BUTTON) == HIGH) { delay(300); // Антидребезг if (!recording) { // Начинаем запись recording = true; stepCount = 0; Serial.println("Запись..."); } else { // Заканчиваем запись и воспроизводим recording = false; Serial.println("Воспроизведение..."); playRecording(); Serial.println("Готово. Нажмите кнопку для новой записи"); } } if (recording && stepCount < 100) { // Записываем текущие позиции for (int i = 0; i < 4; i++) { positions[stepCount][i] = servos[i].read(); } stepCount++; delay(50); } } void playRecording() { for (int step = 0; step < stepCount; step++) { for (int i = 0; i < 4; i++) { servos[i].write(positions[step][i]); } delay(50); } }
🎥 ОБУЧЕНИЕ РОБОТА: Этот метод называется "playback" или "обучение показом". Оператор вручную двигает руку (через потенциометры или просто руками), робот запоминает траекторию и повторяет. Так учат промышленных роботов на заводах .

// ЭКСПЕРИМЕНТ 6: Рука с клешнёй // Сервоприводы: основание, плечо, предплечье, запястье, клешня #include <Servo.h> Servo base; // D9 Servo shoulder; // D10 Servo elbow; // D11 Servo wrist; // D6 Servo gripper; // D5 void setup() { base.attach(9); shoulder.attach(10); elbow.attach(11); wrist.attach(6); gripper.attach(5); // Начальное положение base.write(90); shoulder.write(90); elbow.write(90); wrist.write(90); gripper.write(30); // Клешня открыта delay(1000); } void loop() { // Тянемся к объекту base.write(120); shoulder.write(60); elbow.write(120); delay(1000); // Закрываем клешню (хватаем) gripper.write(80); delay(500); // Поднимаем и поворачиваем shoulder.write(30); base.write(30); delay(1000); // Открываем клешню (кладём) gripper.write(30); delay(500); // Возврат в исходное base.write(90); shoulder.write(90); elbow.write(90); wrist.write(90); delay(1000); }
✋ ТИПЫ ЗАХВАТОВ: В промышленности используют пневматические (сильнее, быстрее), вакуумные (для плоских предметов) и магнитные (для металлов). Сервоприводные клешни — самые простые, но требуют точного управления усилием (чтобы не раздавить объект).

📌 ЭКСПЕРИМЕНТ 7 · СЕМЬ СЕРВОПРИВОДОВ (ГУМАНОИДНАЯ РУКА)

Полная копия человеческой руки (7 степеней свободы)

// ЭКСПЕРИМЕНТ 7: Гуманоидная рука (7 DOF) // Плечевой сустав: 2 серво (поворот/подъём) // Локтевой сустав: 1 серво // Запястье: 2 серво (поворот/наклон) // Клешня: 2 серво (пальцы) #include <Servo.h> // Плечо Servo shoulderRot; // D9 (поворот) Servo shoulderLift; // D10 (подъём) // Локоть Servo elbow; // D11 // Запястье Servo wristRot; // D6 (поворот) Servo wristTilt; // D5 (наклон) // Пальцы Servo thumb; // D3 Servo indexFinger; // D4 int angles[7]; // Текущие углы void setup() { shoulderRot.attach(9); shoulderLift.attach(10); elbow.attach(11); wristRot.attach(6); wristTilt.attach(5); thumb.attach(3); indexFinger.attach(4); // Начальное положение (рука вытянута) setAllAngles(90, 90, 90, 90, 90, 30, 30); delay(1000); } void loop() { // Движение 1: рука вверх setAllAngles(90, 30, 150, 90, 90, 30, 30); delay(1000); // Движение 2: рука в сторону setAllAngles(150, 90, 90, 90, 90, 30, 30); delay(1000); // Движение 3: поворот запястья setAllAngles(90, 90, 90, 30, 150, 30, 30); delay(1000); // Движение 4: взять предмет setAllAngles(120, 60, 120, 90, 90, 80, 80); // Пальцы сжаты delay(1500); // Движение 5: положить предмет setAllAngles(60, 30, 150, 90, 90, 80, 80); delay(1000); setAllAngles(60, 30, 150, 90, 90, 30, 30); // Пальцы разжаты delay(500); // Возврат в исходное setAllAngles(90, 90, 90, 90, 90, 30, 30); delay(1000); } void setAllAngles(int sr, int sl, int e, int wr, int wt, int t, int i) { shoulderRot.write(sr); shoulderLift.write(sl); elbow.write(e); wristRot.write(wr); wristTilt.write(wt); thumb.write(t); indexFinger.write(i); delay(500); // Ждём, пока рука примет положение }
🦾 7 СТЕПЕНЕЙ СВОБОДЫ: Человеческая рука имеет 27 степеней свободы! Но для большинства задач достаточно 7: 3 в плече, 1 в локте, 3 в запястье . Такая рука может выполнять сложные манипуляции — например, собирать мелкие детали.
🤖 ПРОМЫШЛЕННЫЕ РОБОТЫ: KUKA, FANUC, ABB используют именно такую кинематику. Их точность — сотые доли миллиметра, грузоподъёмность — до тонны. Стоимость — от 20 000 до 500 000 евро . Наш сервоприводный вариант — идеальный учебный прототип.
